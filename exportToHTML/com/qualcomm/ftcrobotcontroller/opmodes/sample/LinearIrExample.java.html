32   package com.qualcomm.ftcrobotcontroller.opmodes.sample; 
33    
34   import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode; 
35   import com.qualcomm.robotcore.hardware.DcMotor; 
36   import com.qualcomm.robotcore.hardware.IrSeekerSensor; 
37    
38   /** 
39    * A simple example of a linear op mode that will approach an IR beacon 
40    */ 
41   public class LinearIrExample extends LinearOpMode { 
42    
43     final static double MOTOR_POWER = 0.15; // Higher values will cause the robot to move faster 
44     final static double HOLD_IR_SIGNAL_STRENGTH = 0.20; // Higher values will cause the robot to follow closer 
45    
46     DcMotor motorRight; 
47     DcMotor motorLeft; 
48    
49     IrSeekerSensor irSeeker; 
50    
51     @Override 
52     public void runOpMode() throws InterruptedException { 
53    
54       // set up the hardware devices we are going to use 
55       irSeeker = hardwareMap.irSeekerSensor.get("ir_seeker"); 
56       motorLeft = hardwareMap.dcMotor.get("motor_1"); 
57       motorRight = hardwareMap.dcMotor.get("motor_2"); 
58    
59       motorLeft.setDirection(DcMotor.Direction.REVERSE); 
60    
61       // wait for the start button to be pressed 
62       waitForStart(); 
63    
64       // wait for the IR Seeker to detect a signal 
65       while (!irSeeker.signalDetected()) { 
66         sleep(1000); 
67       } 
68    
69       if (irSeeker.getAngle() < 0) { 
70         // if the signal is to the left move left 
71         motorRight.setPower(MOTOR_POWER); 
72         motorLeft.setPower(-MOTOR_POWER); 
73       } else if (irSeeker.getAngle() > 0) { 
74         // if the signal is to the right move right 
75         motorRight.setPower(-MOTOR_POWER); 
76         motorLeft.setPower(MOTOR_POWER); 
77       } 
78    
79       // wait for the robot to center on the beacon 
80       while (irSeeker.getAngle() != 0) { 
81         waitOneFullHardwareCycle(); 
82       } 
83    
84       // now approach the beacon 
85       motorRight.setPower(MOTOR_POWER); 
86       motorLeft.setPower(MOTOR_POWER); 
87    
88       // wait until we are close enough 
89       while (irSeeker.getStrength() < HOLD_IR_SIGNAL_STRENGTH) { 
90         waitOneFullHardwareCycle(); 
91       } 
92    
93       // stop the motors 
94       motorRight.setPower(0); 
95       motorLeft.setPower(0); 
96     } 
97   }
