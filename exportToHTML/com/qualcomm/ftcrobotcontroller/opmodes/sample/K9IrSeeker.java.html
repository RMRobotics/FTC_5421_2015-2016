32   package com.qualcomm.ftcrobotcontroller.opmodes.sample; 
33    
34   import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
35   import com.qualcomm.robotcore.hardware.DcMotor; 
36   import com.qualcomm.robotcore.hardware.IrSeekerSensor; 
37   import com.qualcomm.robotcore.hardware.Servo; 
38    
39   /** 
40    * TeleOp5421 Mode 
41    * <p> 
42    * Enables control of the robot via the gamepad 
43    */ 
44   public class K9IrSeeker extends OpMode { 
45    
46       final static double MOTOR_POWER = 0.15; // Higher values will cause the robot to move faster 
47       final static double HOLD_IR_SIGNAL_STRENGTH = 0.50; // Higher values will cause the robot to follow closer 
48    
49       double armPosition; 
50       double clawPosition; 
51    
52       DcMotor motorRight; 
53       DcMotor motorLeft; 
54       Servo claw; 
55       Servo arm; 
56       IrSeekerSensor irSeeker; 
57    
58       /** 
59        * Constructor 
60        */ 
61       public K9IrSeeker() { 
62    
63       } 
64    
65       /* 
66        * Code to run when the op mode is first enabled goes here 
67        * 
68        * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#start() 
69        */ 
70       @Override 
71       public void init() { 
72    
73           /* 
74            * Use the hardwareMap to get the dc motors and servos by name. 
75            * Note that the names of the devices must match the names used 
76            * when you configured your robot and created the configuration file. 
77            */ 
78    
79           /* 
80            * For the demo Tetrix K9 bot we assume the following, 
81            *   There are two motors "motor_1" and "motor_2" 
82            *   "motor_1" is on the right side of the bot. 
83            *   "motor_2" is on the left side of the bot. 
84            * 
85            * We also assume that there are two servos "servo_1" and "servo_6" 
86            *    "servo_1" controls the arm joint of the manipulator. 
87            *    "servo_6" controls the claw joint of the manipulator. 
88            */ 
89           motorRight = hardwareMap.dcMotor.get("motor_2"); 
90           motorLeft = hardwareMap.dcMotor.get("motor_1"); 
91           motorLeft.setDirection(DcMotor.Direction.REVERSE); 
92    
93           arm = hardwareMap.servo.get("servo_1"); 
94           claw = hardwareMap.servo.get("servo_6"); 
95    
96           // set the starting position of the wrist and claw 
97           armPosition = 0.1; 
98           clawPosition = 0.25; 
99    
100          /* 
101           * We also assume that we have a Hitechnic IR Seeker v2 sensor 
102           * with a name of "ir_seeker" configured for our robot. 
103           */ 
104          irSeeker = hardwareMap.irSeekerSensor.get("ir_seeker"); 
105      } 
106   
107      /* 
108       * This method will be called repeatedly in a loop 
109       * 
110       * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#run() 
111       */ 
112      @Override 
113      public void loop() { 
114          double angle = 0.0; 
115          double strength = 0.0; 
116          double left, right = 0.0; 
117   
118          // keep manipulator out of the way. 
119          arm.setPosition(armPosition); 
120          claw.setPosition(clawPosition); 
121   
122          /* 
123           * Do we detect an IR signal? 
124           */ 
125          if (irSeeker.signalDetected())  { 
126              /* 
127               * Signal was detected. Follow it. 
128               */ 
129   
130              /* 
131               * Get angle and strength of the signal. 
132               * Note an angle of zero implies straight ahead. 
133               * A negative angle implies that the source is to the left. 
134               * A positive angle implies that the source is to the right. 
135               */ 
136              angle = irSeeker.getAngle(); 
137              strength = irSeeker.getStrength(); 
138   
139              if (angle < -60)  { 
140                  /* 
141                   * IR source is to the way left. 
142                   * Point turn to the left. 
143                   */ 
144                  left = -MOTOR_POWER; 
145                  right = MOTOR_POWER; 
146   
147              } else if (angle < -5) { 
148                  // turn to the left and move forward. 
149                  left = MOTOR_POWER - 0.05; 
150                  right = MOTOR_POWER; 
151              } else if (angle > 5 && angle < 60) { 
152                  // turn to the right and move forward. 
153                  left = MOTOR_POWER; 
154                  right = MOTOR_POWER - 0.05; 
155              } else if (angle > 60) { 
156                  // point turn to right. 
157                  left = MOTOR_POWER; 
158                  right = -MOTOR_POWER; 
159              } else if (strength < HOLD_IR_SIGNAL_STRENGTH) { 
160                  /* 
161                   * Signal is dead ahead but weak. 
162                   * Move forward towards signal 
163                   */ 
164                  left = MOTOR_POWER; 
165                  right = MOTOR_POWER; 
166              } else { 
167                  /* 
168                   * Signal is dead ahead and strong. 
169                   * Stop motors. 
170                   */ 
171                  left = 0.0; 
172                  right = 0.0; 
173              } 
174          } else { 
175              /* 
176               * Signal was not detected. 
177               * Shut off motors 
178               */ 
179              left = 0.0; 
180              right = 0.0; 
181          } 
182   
183          /* 
184           * set the motor power 
185           */ 
186          motorRight.setPower(right); 
187          motorLeft.setPower(left); 
188   
189          /* 
190           * Send telemetry data back to driver station. Note that if we are using 
191           * a legacy NXT-compatible motor controller, then the getPower() method 
192           * will return a null value. The legacy NXT-compatible motor controllers 
193           * are currently write only. 
194           */ 
195   
196          telemetry.addData("Text", "*** Robot Data***"); 
197          telemetry.addData("angle", "angle:  " + Double.toString(angle)); 
198          telemetry.addData("strength", "sig strength: " + Double.toString(strength)); 
199          telemetry.addData("left tgt pwr",  "left  pwr: " + Double.toString(left)); 
200          telemetry.addData("right tgt pwr", "right pwr: " + Double.toString(right)); 
201      } 
202   
203      /* 
204       * Code to run when the op mode is first disabled goes here 
205       * 
206       * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#stop() 
207       */ 
208      @Override 
209      public void stop() { 
210   
211      } 
212   
213  }
