32   package com.qualcomm.ftcrobotcontroller.opmodes.sample; 
33    
34   import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
35   import com.qualcomm.robotcore.hardware.DcMotor; 
36   import com.qualcomm.robotcore.hardware.DcMotorController; 
37   import com.qualcomm.robotcore.hardware.Servo; 
38   import com.qualcomm.robotcore.util.Range; 
39    
40   /** 
41    * TeleOp5421 Mode 
42    * <p> 
43    * Enables control of the robot via the gamepad 
44    */ 
45   public class NxtTeleOp extends OpMode { 
46    
47     // position of the claw servo 
48     double clawPosition; 
49    
50     // amount to change the claw servo position by 
51     double clawDelta = 0.01; 
52    
53     // position of the wrist servo 
54     double wristPosition; 
55    
56     // amount to change the wrist servo position by 
57     double wristDelta = 0.01; 
58    
59     DcMotorController.DeviceMode devMode; 
60     DcMotorController wheelController; 
61     DcMotor motorRight; 
62     DcMotor motorLeft; 
63    
64     Servo claw; 
65     Servo wrist; 
66    
67     int numOpLoops = 1; 
68    
69     /* 
70      * Code to run when the op mode is first enabled goes here 
71      * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#init() 
72      */ 
73     @Override 
74     public void init() { 
75       motorRight = hardwareMap.dcMotor.get("motor_2"); 
76       motorLeft = hardwareMap.dcMotor.get("motor_1"); 
77       claw = hardwareMap.servo.get("servo_6"); // channel 6 
78       wrist = hardwareMap.servo.get("servo_1"); // channel 1 
79    
80       wheelController = hardwareMap.dcMotorController.get("wheels"); 
81     } 
82    
83     /* 
84      * Code that runs repeatedly when the op mode is first enabled goes here 
85      * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#init_loop() 
86      */ 
87     @Override 
88     public void init_loop() { 
89    
90       devMode = DcMotorController.DeviceMode.WRITE_ONLY; 
91    
92       motorRight.setDirection(DcMotor.Direction.REVERSE); 
93       //motorLeft.setDirection(DcMotor.Direction.REVERSE); 
94    
95       // set the mode 
96       // Nxt devices start up in "write" mode by default, so no need to switch device modes here. 
97       motorLeft.setMode(DcMotorController.RunMode.RUN_WITHOUT_ENCODERS); 
98       motorRight.setMode(DcMotorController.RunMode.RUN_WITHOUT_ENCODERS); 
99    
100      wristPosition = 0.6; 
101      clawPosition = 0.5; 
102    } 
103   
104    /* 
105     * This method will be called repeatedly in a loop 
106     * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#loop() 
107     */ 
108    @Override 
109    public void loop() { 
110   
111      // The op mode should only use "write" methods (setPower, setMode, etc) while in 
112      // WRITE_ONLY mode or SWITCHING_TO_WRITE_MODE 
113      if (allowedToWrite()) { 
114      /* 
115       * Gamepad 1 
116       * 
117       * Gamepad 1 controls the motors via the left stick, and it controls the wrist/claw via the a,b, 
118       * x, y buttons 
119       */ 
120   
121        if (gamepad1.dpad_left) { 
122          // Nxt devices start up in "write" mode by default, so no need to switch modes here. 
123          motorLeft.setMode(DcMotorController.RunMode.RUN_WITHOUT_ENCODERS); 
124          motorRight.setMode(DcMotorController.RunMode.RUN_WITHOUT_ENCODERS); 
125        } 
126        if (gamepad1.dpad_right) { 
127          // Nxt devices start up in "write" mode by default, so no need to switch modes here. 
128          motorLeft.setMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
129          motorRight.setMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
130        } 
131   
132        // throttle:  left_stick_y ranges from -1 to 1, where -1 is full up,  and 1 is full down 
133        // direction: left_stick_x ranges from -1 to 1, where -1 is full left and 1 is full right 
134        float throttle = -gamepad1.left_stick_y; 
135        float direction = gamepad1.left_stick_x; 
136        float right = throttle - direction; 
137        float left = throttle + direction; 
138   
139        // clip the right/left values so that the values never exceed +/- 1 
140        right = Range.clip(right, -1, 1); 
141        left = Range.clip(left, -1, 1); 
142   
143        // write the values to the motors 
144        motorRight.setPower(right); 
145        motorLeft.setPower(left); 
146   
147        // update the position of the wrist 
148        if (gamepad1.a) { 
149          wristPosition -= wristDelta; 
150        } 
151   
152        if (gamepad1.y) { 
153          wristPosition += wristDelta; 
154        } 
155   
156        // update the position of the claw 
157        if (gamepad1.x) { 
158          clawPosition -= clawDelta; 
159        } 
160   
161        if (gamepad1.b) { 
162          clawPosition += clawDelta; 
163        } 
164   
165        // clip the position values so that they never exceed 0..1 
166        wristPosition = Range.clip(wristPosition, 0, 1); 
167        clawPosition = Range.clip(clawPosition, 0, 1); 
168   
169        // write position values to the wrist and claw servo 
170        wrist.setPosition(wristPosition); 
171        claw.setPosition(clawPosition); 
172   
173      /* 
174       * Gamepad 2 
175       * 
176       * Gamepad controls the motors via the right trigger as a throttle, left trigger as reverse, and 
177       * the left stick for direction. This type of control is sometimes referred to as race car mode. 
178       */ 
179   
180        // we only want to process gamepad2 if someone is using one of it's analog inputs. If you always 
181        // want to process gamepad2, remove this check 
182        if (gamepad2.atRest() == false) { 
183   
184          // throttle is taken directly from the right trigger, the right trigger ranges in values from 
185          // 0 to 1 
186          throttle = gamepad2.right_trigger; 
187   
188          // if the left trigger is pressed, go in reverse 
189          if (gamepad2.left_trigger != 0.0) { 
190            throttle = -gamepad2.left_trigger; 
191          } 
192   
193          // assign throttle to the left and right motors 
194          right = throttle; 
195          left = throttle; 
196   
197          // now we need to apply steering (direction). The left stick ranges from -1 to 1. If it is 
198          // negative we want to slow down the left motor. If it is positive we want to slow down the 
199          // right motor. 
200          if (gamepad2.left_stick_x < 0) { 
201            // negative value, stick is pulled to the left 
202            left = left * (1 + gamepad2.left_stick_x); 
203          } 
204          if (gamepad2.left_stick_x > 0) { 
205            // positive value, stick is pulled to the right 
206            right = right * (1 - gamepad2.left_stick_x); 
207          } 
208   
209          // write the values to the motor. This will over write any values placed while processing gamepad1 
210          motorRight.setPower(right); 
211          motorLeft.setPower(left); 
212        } 
213      } 
214   
215      // To read any values from the NXT controllers, we need to switch into READ_ONLY mode. 
216      // It takes time for the hardware to switch, so you can't switch modes within one loop of the 
217      // op mode. Every 17th loop, this op mode switches to READ_ONLY mode, and gets the current power. 
218      if (numOpLoops % 17 == 0){ 
219        // Note: If you are using the NxtDcMotorController, you need to switch into "read" mode 
220        // before doing a read, and into "write" mode before doing a write. This is because 
221        // the NxtDcMotorController is on the I2C interface, and can only do one at a time. If you are 
222        // using the USBDcMotorController, there is no need to switch, because USB can handle reads 
223        // and writes without changing modes. The NxtDcMotorControllers start up in "write" mode. 
224        // This method does nothing on USB devices, but is needed on Nxt devices. 
225        wheelController.setMotorControllerDeviceMode(DcMotorController.DeviceMode.READ_ONLY); 
226      } 
227   
228      // Every 17 loops, switch to read mode so we can read data from the NXT device. 
229      // Only necessary on NXT devices. 
230      if (wheelController.getMotorControllerDeviceMode() == DcMotorController.DeviceMode.READ_ONLY) { 
231   
232        // Update the reads after some loops, when the command has successfully propagated through. 
233        telemetry.addData("Text", "free flow text"); 
234        telemetry.addData("left motor", motorLeft.getPower()); 
235        telemetry.addData("right motor", motorRight.getPower()); 
236        telemetry.addData("RunMode: ", motorLeft.getMode().toString()); 
237   
238        // Only needed on Nxt devices, but not on USB devices 
239        wheelController.setMotorControllerDeviceMode(DcMotorController.DeviceMode.WRITE_ONLY); 
240   
241        // Reset the loop 
242        numOpLoops = 0; 
243      } 
244   
245      // Update the current devMode 
246      devMode = wheelController.getMotorControllerDeviceMode(); 
247      numOpLoops++; 
248    } 
249   
250    // If the device is in either of these two modes, the op mode is allowed to write to the HW. 
251    private boolean allowedToWrite(){ 
252      return (devMode == DcMotorController.DeviceMode.WRITE_ONLY); 
253    } 
254  }
