32   package com.qualcomm.ftcrobotcontroller.opmodes.sample; 
33    
34   import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
35   import com.qualcomm.robotcore.hardware.DcMotor; 
36   import com.qualcomm.robotcore.hardware.Servo; 
37   import com.qualcomm.robotcore.util.Range; 
38    
39   /** 
40    * TeleOp5421 Mode 
41    * <p> 
42    * Enables control of the robot via the gamepad 
43    */ 
44   public class K9TeleOp extends OpMode { 
45    
46       /* 
47        * Note: the configuration of the servos is such that 
48        * as the arm servo approaches 0, the arm position moves up (away from the floor). 
49        * Also, as the claw servo approaches 0, the claw opens up (drops the game element). 
50        */ 
51       // TETRIX VALUES. 
52       final static double ARM_MIN_RANGE  = 0.20; 
53       final static double ARM_MAX_RANGE  = 0.90; 
54       final static double CLAW_MIN_RANGE  = 0.20; 
55       final static double CLAW_MAX_RANGE  = 0.7; 
56    
57       // position of the arm servo. 
58       double armPosition; 
59    
60       // amount to change the arm servo position. 
61       double armDelta = 0.1; 
62    
63       // position of the claw servo 
64       double clawPosition; 
65    
66       // amount to change the claw servo position by 
67       double clawDelta = 0.1; 
68    
69       DcMotor motorRight; 
70       DcMotor motorLeft; 
71       Servo claw; 
72       Servo arm; 
73    
74       /** 
75        * Constructor 
76        */ 
77       public K9TeleOp() { 
78    
79       } 
80    
81       /* 
82        * Code to run when the op mode is first enabled goes here 
83        * 
84        * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#start() 
85        */ 
86       @Override 
87       public void init() { 
88    
89    
90           /* 
91            * Use the hardwareMap to get the dc motors and servos by name. Note 
92            * that the names of the devices must match the names used when you 
93            * configured your robot and created the configuration file. 
94            */ 
95    
96           /* 
97            * For the demo Tetrix K9 bot we assume the following, 
98            *   There are two motors "motor_1" and "motor_2" 
99            *   "motor_1" is on the right side of the bot. 
100           *   "motor_2" is on the left side of the bot and reversed. 
101           * 
102           * We also assume that there are two servos "servo_1" and "servo_6" 
103           *    "servo_1" controls the arm joint of the manipulator. 
104           *    "servo_6" controls the claw joint of the manipulator. 
105           */ 
106          motorRight = hardwareMap.dcMotor.get("motor_2"); 
107          motorLeft = hardwareMap.dcMotor.get("motor_1"); 
108          motorLeft.setDirection(DcMotor.Direction.REVERSE); 
109   
110          arm = hardwareMap.servo.get("servo_1"); 
111          claw = hardwareMap.servo.get("servo_6"); 
112   
113          // assign the starting position of the wrist and claw 
114          armPosition = 0.2; 
115          clawPosition = 0.2; 
116      } 
117   
118      /* 
119       * This method will be called repeatedly in a loop 
120       * 
121       * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#run() 
122       */ 
123      @Override 
124      public void loop() { 
125   
126          /* 
127           * Gamepad 1 
128           * 
129           * Gamepad 1 controls the motors via the left stick, and it controls the 
130           * wrist/claw via the a,b, x, y buttons 
131           */ 
132   
133          // throttle: left_stick_y ranges from -1 to 1, where -1 is full up, and 
134          // 1 is full down 
135          // direction: left_stick_x ranges from -1 to 1, where -1 is full left 
136          // and 1 is full right 
137          float throttle = -gamepad1.left_stick_y; 
138          float direction = gamepad1.left_stick_x; 
139          float right = throttle - direction; 
140          float left = throttle + direction; 
141   
142          // clip the right/left values so that the values never exceed +/- 1 
143          right = Range.clip(right, -1, 1); 
144          left = Range.clip(left, -1, 1); 
145   
146          // scale the joystick value to make it easier to control 
147          // the robot more precisely at slower speeds. 
148          right = (float)scaleInput(right); 
149          left =  (float)scaleInput(left); 
150   
151          // write the values to the motors 
152          motorRight.setPower(right); 
153          motorLeft.setPower(left); 
154   
155          // update the position of the arm. 
156          if (gamepad1.a) { 
157              // if the A button is pushed on gamepad1, increment the position of 
158              // the arm servo. 
159              armPosition += armDelta; 
160          } 
161   
162          if (gamepad1.y) { 
163              // if the Y button is pushed on gamepad1, decrease the position of 
164              // the arm servo. 
165              armPosition -= armDelta; 
166          } 
167   
168          // update the position of the claw 
169          if (gamepad1.x) { 
170              clawPosition += clawDelta; 
171          } 
172   
173          if (gamepad1.b) { 
174              clawPosition -= clawDelta; 
175          } 
176   
177          // clip the position values so that they never exceed their allowed range. 
178          armPosition = Range.clip(armPosition, ARM_MIN_RANGE, ARM_MAX_RANGE); 
179          clawPosition = Range.clip(clawPosition, CLAW_MIN_RANGE, CLAW_MAX_RANGE); 
180   
181          // write position values to the wrist and claw servo 
182          arm.setPosition(armPosition); 
183          claw.setPosition(clawPosition); 
184   
185   
186   
187          /* 
188           * Send telemetry data back to driver station. Note that if we are using 
189           * a legacy NXT-compatible motor controller, then the getPower() method 
190           * will return a null value. The legacy NXT-compatible motor controllers 
191           * are currently write only. 
192           */ 
193          telemetry.addData("Text", "*** Robot Data***"); 
194          telemetry.addData("arm", "arm:  " + String.format("%.2f", armPosition)); 
195          telemetry.addData("claw", "claw:  " + String.format("%.2f", clawPosition)); 
196          telemetry.addData("left tgt pwr",  "left  pwr: " + String.format("%.2f", left)); 
197          telemetry.addData("right tgt pwr", "right pwr: " + String.format("%.2f", right)); 
198   
199      } 
200   
201      /* 
202       * Code to run when the op mode is first disabled goes here 
203       * 
204       * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#stop() 
205       */ 
206      @Override 
207      public void stop() { 
208   
209      } 
210   
211   
212      /* 
213       * This method scales the joystick input so for low joystick values, the 
214       * scaled value is less than linear.  This is to make it easier to drive 
215       * the robot more precisely at slower speeds. 
216       */ 
217      double scaleInput(double dVal)  { 
218          double[] scaleArray = { 0.0, 0.05, 0.09, 0.10, 0.12, 0.15, 0.18, 0.24, 
219                  0.30, 0.36, 0.43, 0.50, 0.60, 0.72, 0.85, 1.00, 1.00 }; 
220   
221          // get the corresponding index for the scaleInput array. 
222          int index = (int) (dVal * 16.0); 
223   
224          // index should be positive. 
225          if (index < 0) { 
226              index = -index; 
227          } 
228   
229          // index cannot exceed size of array minus 1. 
230          if (index > 16) { 
231              index = 16; 
232          } 
233   
234          // get value from the array. 
235          double dScale = 0.0; 
236          if (dVal < 0) { 
237              dScale = -scaleArray[index]; 
238          } else { 
239              dScale = scaleArray[index]; 
240          } 
241   
242          // return scaled value. 
243          return dScale; 
244      } 
245   
246  }
