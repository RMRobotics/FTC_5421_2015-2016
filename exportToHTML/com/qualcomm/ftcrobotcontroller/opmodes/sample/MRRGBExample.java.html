32   package com.qualcomm.ftcrobotcontroller.opmodes.sample; 
33    
34   import android.app.Activity; 
35   import android.graphics.Color; 
36   import android.view.View; 
37    
38   import com.qualcomm.ftccommon.DbgLog; 
39   import com.qualcomm.ftcrobotcontroller.R; 
40   import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode; 
41   import com.qualcomm.robotcore.hardware.ColorSensor; 
42    
43   /* 
44    * 
45    * This is an example LinearOpMode that shows how to use 
46    * a Modern Robotics Color Sensor. 
47    * 
48    * The op mode assumes that the color sensor 
49    * is configured with a name of "mr". 
50    * 
51    * You can use the X button on either gamepad to turn the LED on and off. 
52    * 
53    */ 
54   public class MRRGBExample extends LinearOpMode { 
55    
56     ColorSensor sensorRGB; 
57    
58    
59     @Override 
60     public void runOpMode() throws InterruptedException { 
61    
62       // write some device information (connection info, name and type) 
63       // to the log file. 
64       hardwareMap.logDevices(); 
65    
66       // get a reference to our ColorSensor object. 
67       sensorRGB = hardwareMap.colorSensor.get("mr"); 
68    
69       // bEnabled represents the state of the LED. 
70       boolean bEnabled = true; 
71    
72       // turn the LED on in the beginning, just so user will know that the sensor is active. 
73       sensorRGB.enableLed(true); 
74    
75       // wait one cycle. 
76       waitOneFullHardwareCycle(); 
77    
78       // wait for the start button to be pressed. 
79       waitForStart(); 
80    
81       // hsvValues is an array that will hold the hue, saturation, and value information. 
82       float hsvValues[] = {0F,0F,0F}; 
83    
84       // values is a reference to the hsvValues array. 
85       final float values[] = hsvValues; 
86    
87       // get a reference to the RelativeLayout so we can change the background 
88       // color of the Robot Controller app to match the hue detected by the RGB sensor. 
89       final View relativeLayout = ((Activity) hardwareMap.appContext).findViewById(R.id.RelativeLayout); 
90    
91       // bPrevState and bCurrState represent the previous and current state of the button. 
92       boolean bPrevState = false; 
93       boolean bCurrState = false; 
94    
95       // while the op mode is active, loop and read the RGB data. 
96       // Note we use opModeIsActive() as our loop condition because it is an interruptible method. 
97       while (opModeIsActive()) { 
98         // check the status of the x button on either gamepad. 
99         bCurrState = gamepad1.x || gamepad2.x; 
100   
101        // check for button state transitions. 
102        if (bCurrState == true && bCurrState != bPrevState)  { 
103          // button is transitioning to a pressed state. 
104   
105          // print a debug statement. 
106          DbgLog.msg("MY_DEBUG - x button was pressed!"); 
107   
108          // update previous state variable. 
109          bPrevState = bCurrState; 
110   
111          // on button press, enable the LED. 
112          bEnabled = true; 
113   
114          // turn on the LED. 
115          sensorRGB.enableLed(bEnabled); 
116        } else if (bCurrState == false && bCurrState != bPrevState)  { 
117          // button is transitioning to a released state. 
118   
119          // print a debug statement. 
120          DbgLog.msg("MY_DEBUG - x button was released!"); 
121   
122          // update previous state variable. 
123          bPrevState = bCurrState; 
124   
125          // on button press, enable the LED. 
126          bEnabled = false; 
127   
128          // turn off the LED. 
129          sensorRGB.enableLed(false); 
130        } 
131   
132        // convert the RGB values to HSV values. 
133        //Color.RGBToHSV((sensorRGB.red() * 8), (sensorRGB.green() * 8), (sensorRGB.blue() * 8), hsvValues); 
134        Color.RGBToHSV(sensorRGB.red()*8, sensorRGB.green()*8, sensorRGB.blue()*8, hsvValues); 
135   
136        // send the info back to driver station using telemetry function. 
137        telemetry.addData("Clear", sensorRGB.alpha()); 
138        telemetry.addData("Red  ", sensorRGB.red()); 
139        telemetry.addData("Green", sensorRGB.green()); 
140        telemetry.addData("Blue ", sensorRGB.blue()); 
141        telemetry.addData("Hue", hsvValues[0]); 
142   
143        // change the background color to match the color detected by the RGB sensor. 
144        // pass a reference to the hue, saturation, and value array as an argument 
145        // to the HSVToColor method. 
146        relativeLayout.post(new Runnable() { 
147          public void run() { 
148            relativeLayout.setBackgroundColor(Color.HSVToColor(0xff, values)); 
149          } 
150        }); 
151   
152        // wait a hardware cycle before iterating. 
153        waitOneFullHardwareCycle(); 
154      } 
155    } 
156  }
