15    */ 
16   public class PushBotManual1 extends PushBotTelemetry 
17    
18   { 
19       //-------------------------------------------------------------------------- 
20       // 
21       // PushBotManual1 
22       // 
23       /** 
24        * Construct the class. 
25        * 
26        * The system calls this member when the class is instantiated. 
27        */ 
28       public PushBotManual1 () 
29    
30       { 
31           // 
32           // Initialize base classes. 
33           // 
34           // All via self-construction. 
35    
36           // 
37           // Initialize class members. 
38           // 
39           // All via self-construction. 
40    
41       } // PushBotManual1

43       //-------------------------------------------------------------------------- 
44       // 
45       // loop 
46       // 
47       /** 
48        * Implement a state machine that controls the robot during 
49        * manual-operation.  The state machine uses gamepad input to transition 
50        * between states. 
51        * 
52        * The system calls this member repeatedly while the OpMode is running. 
53        */ 
54       @Override public void loop () 
55    
56       { 
57           //---------------------------------------------------------------------- 
58           // 
59           // DC Motors 
60           // 
61           // Obtain the current values of the joystick controllers. 
62           // 
63           // Note that x and y equal -1 when the joystick is pushed all of the way 
64           // forward (i.e. away from the human holder's body). 
65           // 
66           // The clip method guarantees the value never exceeds the range +-1. 
67           // 
68           // The DC motors are scaled to make it easier to control them at slower 
69           // speeds. 
70           // 
71           // The setPower methods write the motor power values to the DcMotor 
72           // class, but the power levels aren't applied until this method ends. 
73           // 
74    
75           // 
76           // Manage the drive wheel motors. 
77           // 
78           float l_gp1_left_stick_y = -gamepad1.left_stick_y; 
79           float l_left_drive_power 
80               = scale_motor_power (l_gp1_left_stick_y); 
81    
82           float l_gp1_right_stick_y = -gamepad1.right_stick_y; 
83           float l_right_drive_power 
84               = scale_motor_power (l_gp1_right_stick_y); 
85    
86           set_drive_power (l_left_drive_power, l_right_drive_power); 
87    
88           // 
89           // Manage the arm motor.  The right trigger makes the arm move from the 
90           // front of the robot to the back (i.e. up).  The left trigger makes the 
91           // arm move from the back to the front (i.e. down). 
92           // 
93           float l_left_arm_power 
94               = scale_motor_power (gamepad1.right_trigger) 
95               - scale_motor_power (gamepad1.left_trigger); 
96           m_left_arm_power (l_left_arm_power); 
97    
98           //---------------------------------------------------------------------- 
99           // 
100          // Servo Motors 
101          // 
102          // Obtain the current values of the gamepad 'x' and 'b' buttons. 
103          // 
104          // Note that x and b buttons have boolean values of true and false. 
105          // 
106          // The clip method guarantees the value never exceeds the allowable 
107          // range of [0,1]. 
108          // 
109          // The setPosition methods write the motor power values to the Servo 
110          // class, but the positions aren't applied until this method ends. 
111          // 
112          if (gamepad1.x) 
113          { 
114              m_hand_position (a_hand_position () + 0.05); 
115          } 
116          else if (gamepad1.b) 
117          { 
118              m_hand_position (a_hand_position () - 0.05); 
119          } 
120   
121          // 
122          // Send telemetry data to the driver station. 
123          // 
124          update_telemetry (); // Update common telemetry 
125          update_gamepad_telemetry (); 
126          telemetry.addData 
127              ( "12" 
128              , "Left Arm: " + l_left_arm_power 
129              ); 
130   
131      } // loop 
132   
133  } // PushBotManual1
