1    package com.qualcomm.ftcrobotcontroller.opmodes.sample;

3    import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode; 
4    import com.qualcomm.robotcore.hardware.DcMotor; 
5    import com.qualcomm.robotcore.hardware.DcMotorController; 
6     
7    /* 
8     * An example linear op mode where the pushbot 
9     * will drive in a square pattern using sleep() 
10    * and a for loop. 
11    */ 
12   public class PushBotSquare extends LinearOpMode { 
13       DcMotor leftMotor; 
14       DcMotor rightMotor; 
15    
16       @Override 
17       public void runOpMode() throws InterruptedException { 
18           leftMotor = hardwareMap.dcMotor.get("left_drive"); 
19           rightMotor = hardwareMap.dcMotor.get("right_drive"); 
20           rightMotor.setDirection(DcMotor.Direction.REVERSE); 
21           leftMotor.setMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
22           rightMotor.setMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
23    
24           waitForStart(); 
25    
26           for(int i=0; i<4; i++) { 
27               leftMotor.setPower(1.0); 
28               rightMotor.setPower(1.0); 
29    
30               sleep(1000); 
31    
32               leftMotor.setPower(0.5); 
33               rightMotor.setPower(-0.5); 
34    
35               sleep(500); 
36           } 
37    
38           leftMotor.setPowerFloat(); 
39           rightMotor.setPowerFloat(); 
40    
41       } 
42   }
