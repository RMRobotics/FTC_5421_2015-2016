1    package com.qualcomm.ftcrobotcontroller.opmodes.sample; 
2     
3    import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode; 
4    import com.qualcomm.robotcore.hardware.DcMotor; 
5    import com.qualcomm.robotcore.hardware.TouchSensor; 
6     
7    /** 
8     * Created by Jordan Burklund on 7/30/2015. 
9     * An example linear op mode where the pushbot 
10    * will run its motors unless a touch sensor 
11    * is pressed. 
12    */ 
13   public class PushBotDriveTouch extends LinearOpMode { 
14       DcMotor leftMotor; 
15       DcMotor rightMotor; 
16       TouchSensor touchSensor; 
17    
18       @Override 
19       public void runOpMode() throws InterruptedException { 
20           // Get references to the motors from the hardware map 
21           leftMotor = hardwareMap.dcMotor.get("left_drive"); 
22           rightMotor = hardwareMap.dcMotor.get("right_drive"); 
23    
24           // Reverse the right motor 
25           rightMotor.setDirection(DcMotor.Direction.REVERSE); 
26    
27           // Get a reference to the touch sensor 
28           touchSensor = hardwareMap.touchSensor.get("sensor_touch"); 
29    
30           // Wait for the start button to be pressed 
31           waitForStart(); 
32    
33           while(opModeIsActive()) { 
34               if(touchSensor.isPressed()) { 
35                   //Stop the motors if the touch sensor is pressed 
36                   leftMotor.setPower(0); 
37                   rightMotor.setPower(0); 
38               } else { 
39                   //Keep driving if the touch sensor is not pressed 
40                   leftMotor.setPower(0.5); 
41                   rightMotor.setPower(0.5); 
42               } 
43    
44               telemetry.addData("isPressed", String.valueOf(touchSensor.isPressed())); 
45    
46               // Wait for a hardware cycle to allow other processes to run 
47               waitOneFullHardwareCycle(); 
48           } 
49    
50       } 
51   }
