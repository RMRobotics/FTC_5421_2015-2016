8    /** 
9     * Provide telemetry provided by the PushBotHardware class. 
10    * 
11    * Insert this class between a custom op-mode and the PushBotHardware class to 
12    * display telemetry available from the hardware class. 
13    * 
14    * @author SSI Robotics 
15    * @version 2015-08-02-13-57 
16    * 
17    * Telemetry Keys 
18    *     00 - Important message; sometimes used for error messages. 
19    *     01 - The power being sent to the left drive's motor controller and the 
20    *          encoder count returned from the motor controller. 
21    *     02 - The power being sent to the right drive's motor controller and the 
22    *          encoder count returned from the motor controller. 
23    *     03 - The power being sent to the left arm's motor controller. 
24    *     04 - The position being sent to the left and right hand's servo 
25    *          controller. 
26    *     05 - The negative value of gamepad 1's left stick's y (vertical; up/down) 
27    *          value. 
28    *     06 - The negative value of gamepad 1's right stick's y (vertical; 
29    *          up/down) value. 
30    *     07 - The negative value of gamepad 2's left stick's y (vertical; up/down) 
31    *          value. 
32    *     08 - The value of gamepad 2's X button (true/false). 
33    *     09 - The value of gamepad 2's Y button (true/false). 
34    *     10 - The value of gamepad 1's left trigger value. 
35    *     11 - The value of gamepad 1's right trigger value. 
36    */ 
37   public class PushBotTelemetry extends PushBotHardware 
38    
39   {

41       // 
42       // PushBotTelemetry 
43       //

45        * Construct the class.

49       public PushBotTelemetry () 
50    
51       { 
52           // 
53           // Initialize base classes. 
54           // 
55           // All via self-construction. 
56    
57           // 
58           // Initialize class members. 
59           // 
60           // All via self-construction. 
61    
62       } // PushBotTelemetry

64       //-------------------------------------------------------------------------- 
65       // 
66       // update_telemetry 
67       // 
68       /** 
69        * Update the telemetry with current values from the base class. 
70        */ 
71       public void update_telemetry () 
72    
73       { 
74           if (a_warning_generated ()) 
75           { 
76               set_first_message (a_warning_message ()); 
77           } 
78           // 
79           // Send telemetry data to the driver station. 
80           // 
81           telemetry.addData 
82               ( "01" 
83               , "Left Drive: " 
84                   + a_left_drive_power () 
85                   + ", " 
86                   + a_left_encoder_count () 
87               ); 
88           telemetry.addData 
89               ( "02" 
90               , "Right Drive: " 
91                   + a_right_drive_power () 
92                   + ", " 
93                   + a_right_encoder_count () 
94               ); 
95           telemetry.addData 
96               ( "03" 
97               , "Left Arm: " + a_left_arm_power () 
98               ); 
99           telemetry.addData 
100              ( "04" 
101              , "Hand Position: " + a_hand_position () 
102              ); 
103   
104      } // update_telemetry 
105   
106      //-------------------------------------------------------------------------- 
107      // 
108      // update_gamepad_telemetry 
109      // 
110      /** 
111       * Update the telemetry with current gamepad readings. 
112       */ 
113      public void update_gamepad_telemetry () 
114   
115      { 
116          // 
117          // Send telemetry data concerning gamepads to the driver station. 
118          // 
119          telemetry.addData ("05", "GP1 Left: " + -gamepad1.left_stick_y); 
120          telemetry.addData ("06", "GP1 Right: " + -gamepad1.right_stick_y); 
121          telemetry.addData ("07", "GP2 Left: " + -gamepad2.left_stick_y); 
122          telemetry.addData ("08", "GP2 X: " + gamepad2.x); 
123          telemetry.addData ("09", "GP2 Y: " + gamepad2.y); 
124          telemetry.addData ("10", "GP1 LT: " + gamepad1.left_trigger); 
125          telemetry.addData ("11", "GP1 RT: " + gamepad1.right_trigger); 
126   
127      } // update_gamepad_telemetry 
128

133      /** 
134       * Update the telemetry's first message with the specified message. 
135       */ 
136      public void set_first_message (String p_message)

140   
141      } // set_first_message

150      public void set_error_message (String p_message)

153          set_first_message ("ERROR: " + p_message); 
154   
155      } // set_error_message
