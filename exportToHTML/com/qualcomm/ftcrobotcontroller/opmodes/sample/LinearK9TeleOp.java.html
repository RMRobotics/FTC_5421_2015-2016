32   package com.qualcomm.ftcrobotcontroller.opmodes.sample; 
33    
34   import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode; 
35   import com.qualcomm.robotcore.hardware.DcMotor; 
36   import com.qualcomm.robotcore.hardware.Servo; 
37   import com.qualcomm.robotcore.util.Range; 
38    
39   /** 
40    * Linear Tele Op Mode 
41    * <p> 
42    * Enables control of the robot via the gamepad. 
43    * NOTE: This op mode will not work with the NXT Motor Controllers. Use an Nxt op mode instead. 
44    */ 
45   public class LinearK9TeleOp extends LinearOpMode { 
46    
47     // position of the neck servo 
48     double neckPosition; 
49     double jawPosition; 
50    
51     // amount to change the neck servo position by 
52     double neckDelta = 0.01; 
53    
54     DcMotor motorRight; 
55     DcMotor motorLeft; 
56    
57     Servo neck; 
58     Servo jaw; 
59    
60     @Override 
61     public void runOpMode() throws InterruptedException { 
62       motorLeft = hardwareMap.dcMotor.get("motor_1"); 
63       motorRight = hardwareMap.dcMotor.get("motor_2"); 
64       neck = hardwareMap.servo.get("servo_1"); 
65       jaw = hardwareMap.servo.get("servo_6"); 
66    
67       motorLeft.setDirection(DcMotor.Direction.REVERSE); 
68    
69       // set the starting position of the wrist and neck 
70       neckPosition = 0.5; 
71    
72       waitForStart(); 
73    
74       while (opModeIsActive()) { 
75         // throttle:  left_stick_y ranges from -1 to 1, where -1 is full up,  and 1 is full down 
76         // direction: left_stick_x ranges from -1 to 1, where -1 is full left and 1 is full right 
77         float throttle  = -gamepad1.left_stick_y; 
78         float direction =  gamepad1.left_stick_x; 
79         float right = throttle - direction; 
80         float left  = throttle + direction; 
81    
82         // clip the right/left values so that the values never exceed +/- 1 
83         right = Range.clip(right, -1, 1); 
84         left  = Range.clip(left,  -1, 1); 
85    
86         // write the values to the motors 
87         motorRight.setPower(right); 
88         motorLeft.setPower(left); 
89    
90         // update the position of the neck 
91         if (gamepad1.y) { 
92           neckPosition -= neckDelta; 
93         } 
94    
95         if (gamepad1.a) { 
96           neckPosition += neckDelta; 
97         } 
98    
99         // clip the position values so that they never exceed 0..1 
100        neckPosition = Range.clip(neckPosition, 0, 1); 
101   
102        // set jaw position 
103        jawPosition = 1 - Range.scale(gamepad1.right_trigger, 0.0, 1.0, 0.3, 1.0); 
104   
105        // write position values to the wrist and neck servo 
106        neck.setPosition(neckPosition); 
107        jaw.setPosition(jawPosition); 
108   
109        telemetry.addData("Text", "K9TeleOp"); 
110        telemetry.addData(" left motor", motorLeft.getPower()); 
111        telemetry.addData("right motor", motorRight.getPower()); 
112        telemetry.addData("neck", neck.getPosition()); 
113        telemetry.addData("jaw", jaw.getPosition()); 
114   
115        waitOneFullHardwareCycle(); 
116      } 
117    } 
118  }
