273      /** 
274       * Scale the joystick input using a nonlinear algorithm. 
275       */ 
276      float scale_motor_power (float p_power) 
277      { 
278          // 
279          // Assume no scaling. 
280          // 
281          float l_scale = 0.0f; 
282   
283          // 
284          // Ensure the values are legal. 
285          // 
286          float l_power = Range.clip (p_power, -1, 1); 
287   
288          float[] l_array = 
289              { 0.00f, 0.05f, 0.09f, 0.10f, 0.12f 
290              , 0.15f, 0.18f, 0.24f, 0.30f, 0.36f 
291              , 0.43f, 0.50f, 0.60f, 0.72f, 0.85f 
292              , 1.00f, 1.00f 
293              }; 
294   
295          // 
296          // Get the corresponding index for the specified argument/parameter. 
297          // 
298          int l_index = (int)(l_power * 16.0); 
299          if (l_index < 0) 
300          { 
301              l_index = -l_index; 
302          } 
303          else if (l_index > 16) 
304          { 
305              l_index = 16; 
306          } 
307   
308          if (l_power < 0) 
309          { 
310              l_scale = -l_array[l_index]; 
311          } 
312          else 
313          { 
314              l_scale = l_array[l_index]; 
315          } 
316   
317          return l_scale; 
318   
319      } // scale_motor_power

759   
760              //
