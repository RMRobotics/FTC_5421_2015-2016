32   package com.qualcomm.ftcrobotcontroller.opmodes.sample; 
33    
34   import com.qualcomm.robotcore.eventloop.opmode.LinearOpMode; 
35   import com.qualcomm.robotcore.hardware.GyroSensor; 
36    
37   /* 
38    * 
39    * This is an example LinearOpMode that shows how to use 
40    * the Modern Robotics Gyro. 
41    * 
42    * The op mode assumes that the gyro sensor 
43    * is configured with a name of "gyro". 
44    * 
45    * 
46    * 
47    */ 
48   public class MRGyroTest extends LinearOpMode { 
49    
50    
51     @Override 
52     public void runOpMode() throws InterruptedException { 
53    
54       GyroSensor sensorGyro; 
55       int xVal, yVal, zVal = 0; 
56       int heading = 0; 
57    
58       // write some device information (connection info, name and type) 
59       // to the log file. 
60       hardwareMap.logDevices(); 
61    
62       // get a reference to our GyroSensor object. 
63       sensorGyro = hardwareMap.gyroSensor.get("gyro"); 
64    
65       // calibrate the gyro. 
66       sensorGyro.calibrate(); 
67    
68       // wait for the start button to be pressed. 
69       waitForStart(); 
70    
71       // make sure the gyro is calibrated. 
72       while (sensorGyro.isCalibrating())  { 
73         Thread.sleep(50); 
74       } 
75    
76       while (opModeIsActive())  { 
77         // if the A and B buttons are pressed, reset Z heading. 
78         if(gamepad1.a && gamepad1.b)  { 
79           // reset heading. 
80           sensorGyro.resetZAxisIntegrator(); 
81         } 
82    
83         // get the x, y, and z values (rate of change of angle). 
84         xVal = sensorGyro.rawX(); 
85         yVal = sensorGyro.rawY(); 
86         zVal = sensorGyro.rawZ(); 
87    
88         // get the heading info. 
89         // the Modern Robotics' gyro sensor keeps 
90         // track of the current heading for the Z axis only. 
91         heading = sensorGyro.getHeading(); 
92    
93         telemetry.addData("1. x", String.format("%03d", xVal)); 
94         telemetry.addData("2. y", String.format("%03d", yVal)); 
95         telemetry.addData("3. z", String.format("%03d", zVal)); 
96         telemetry.addData("4. h", String.format("%03d", heading)); 
97    
98         Thread.sleep(100); 
99       } 
100    } 
101  }
