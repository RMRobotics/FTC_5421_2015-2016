1    package com.qualcomm.ftcrobotcontroller.opmodes.custom.util; 
2     
3    import com.rmrobotics.library.control.Axis; 
4    import com.rmrobotics.library.control.Controller; 
5    import com.rmrobotics.library.control.Joystick; 
6    import com.rmrobotics.library.core.RMOpMode; 
7     
8    public class ServoCalibration extends RMOpMode { 
9     
10       @Override 
11       public void init() { 
12           super.setTeam(5421); 
13           super.init(); 
14       } 
15    
16       @Override 
17       protected void calculate() { 
18           double leftFlap = control.joystickValue(Controller.C_TWO, Joystick.J_LEFT, Axis.A_Y); 
19           double rightFlap = control.joystickValue(Controller.C_TWO, Joystick.J_RIGHT, Axis.A_Y); 
20           double lFlapPos = servoMap.get("bL").getPosition(); 
21           double rFlapPos = servoMap.get("bR").getPosition(); 
22           if (leftFlap > 0.2) { 
23               lFlapPos += 0.01; 
24               servoMap.get("bL").setDesiredPosition(lFlapPos); 
25           } else if (leftFlap < -0.2) { 
26               lFlapPos -= 0.01; 
27               servoMap.get("bL").setDesiredPosition(lFlapPos); 
28           } 
29           if (rightFlap > 0.2) { 
30               rFlapPos += 0.01; 
31               servoMap.get("bR").setDesiredPosition(rFlapPos); 
32           } else if (rightFlap < -0.2) { 
33               rFlapPos -= 0.01; 
34               servoMap.get("bR").setDesiredPosition(rFlapPos); 
35           } 
36           telemetry.addData("L-LJ-R-RJ", lFlapPos + " " + leftFlap + " " + rFlapPos + " " + rightFlap); 
37       } 
38   }
