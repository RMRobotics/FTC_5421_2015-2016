13   public class GyroTest5421 extends OpMode { 
14    
15       GyroSensor gyro; 
16       DcMotor mL; 
17       DcMotor mR; 
18       Gamepad g1; 
19    
20       @Override 
21       public void init() { 
22           gyro = hardwareMap.gyroSensor.get("gyro"); 
23           gyro.calibrate(); 
24           mL = hardwareMap.dcMotor.get("mL"); 
25           mL.setMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
26           mR = hardwareMap.dcMotor.get("mR"); 
27           mR.setMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
28           mR.setDirection(DcMotor.Direction.REVERSE); 
29           g1 = gamepad1; 
30       } 
31    
32       public void init_loop() { 
33           telemetry.addData("GYRO CALIBRATING:", gyro.isCalibrating()); 
34       } 
35    
36       @Override 
37       public void loop() { 
38           double leftPower = g1.left_stick_y; 
39           double rightPower = g1.right_stick_y; 
40           if (Math.abs(45-gyro.getHeading()) > 3) { 
41               mL.setPower(leftPower * 0.2); 
42               mR.setPower(rightPower * 0.2); 
43           } else { 
44               mL.setPower(0); 
45               mR.setPower(0); 
46           } 
47    
48    
49           telemetry.addData("h", String.format("%03d", gyro.getHeading())); 
50           telemetry.addData("t", String.format("%05d", getRuntime())); 
51           telemetry.addData("mL", mL.getPower()); 
52           telemetry.addData("mR", mR.getPower()); 
53       } 
54   }
