40       @Override 
41       public void init() { 
42           super.setTeam(5421); 
43           super.init(); 
44           gyro = hardwareMap.gyroSensor.get("gyro"); 
45           gyro.calibrate(); 
46           mL = motorMap.get("mL"); 
47           sL = motorMap.get("sL"); 
48           mR = motorMap.get("mR"); 
49           sR = motorMap.get("sR"); 
50           wL = motorMap.get("wL"); 
51           wR = motorMap.get("wR"); 
52           h = motorMap.get("h"); 
53           //climbers = servoMap.get("climbers");

59       @Override 
60       protected void calculate() { 
61           for (Motor m : motorMap.values()) { 
62               if (m.getMode() != DcMotorController.RunMode.RUN_USING_ENCODERS) { 
63                   m.setMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
64               } 
65           } 
66           switch (state) { 
67               case 1: 
68                   setDriveTarget(10000); 
69                   setDrivePower(0.5); 
70                   updateDP(); 
71                   updateState(StateType.GYRO_ENCODER_DRIVE); 
72                   break; 
73               default: 
74                   stop(); 
75                   break; 
76               case WAIT: 
77                   switch (prevStateType) { 
78                       case ENCODER_DRIVE: 
79                           if (driveDone()) { 
80                               updateStateWait(); 
81                               driveStop(); 
82                           } 
83                           break; 
84                       case GYRO_ENCODER_DRIVE: 
85                           if (gyroEncoderDriveDone()) { 
86                               updateStateWait(); 
87                               driveStop(); 
88                           } 
89                           break; 
90                       case GYRO_DRIVE: 
91                           if (gyroTimeDriveDone()) { 
92                               updateStateWait(); 
93                               driveStop(); 
94                           } 
95                           break; 
96                       case SLEEP: 
97                           if (timeDone()) { 
98                               updateStateWait(); 
99                           } 
100                          break; 
101                      case GYRO: 
102                          if (gyroDone()) { 
103                              updateStateWait(); 
104                          } 
105                          break; 
106                  } 
107          } 
108          updateSlave(); 
109          addTelemetry(); 
110      } 
111   
112      private void addTelemetry() { 
113          telemetry.addData("S-P", state); 
114          telemetry.addData("ML-LT-LP", df.format(mL.getPower()) + "-" + nf.format(mL.getTargetPosition()) + "-" + nf.format(mL.getCurrentPosition())); 
115          telemetry.addData("MR-RT-RP", df.format(mR.getPower()) + "-" + nf.format(mR.getTargetPosition()) + "-" + nf.format(mR.getCurrentPosition())); 
116          telemetry.addData("H", df.format(h.getPower())); 
117          telemetry.addData("WL", df.format(wL.getPower())); 
118          telemetry.addData("WR", df.format(wR.getPower())); 
119          telemetry.addData("GYRO", gyro.getHeading() + "-" + skew); 
120          telemetry.addData("TIME", runTime.time()); 
121      }

123      private void setDriveTarget(int target) { 
124          mL.setTargetPosition(target); 
125          mR.setTargetPosition(target); 
126      } 
127   
128      private void setDriveTarget(int targetLeft, int targetRight) { 
129          mL.setTargetPosition(targetLeft); 
130          mR.setTargetPosition(targetRight); 
131      } 
132   
133      private void setDrivePower(double power) { 
134          mL.setDesiredPower(power); 
135          mR.setDesiredPower(power); 
136      } 
137   
138      private void setDrivePower(double powerLeft, double powerRight) { 
139          mL.setDesiredPower(powerLeft); 
140          mR.setDesiredPower(powerRight); 
141      } 
142   
143      private void updateState(StateType type) { 
144          prevState = state; 
145          prevStateType = type; 
146          state = WAIT; 
147      } 
148   
149      private void updateStateWait() { 
150          state = prevState + 1; 
151          prevState = WAIT; 
152          prevStateType = StateType.WAIT; 
153      } 
154   
155      private void updateSlave() { 
156          sL.setDesiredPower(mL.getDesiredPower()); 
157          sR.setDesiredPower(mR.getDesiredPower()); 
158      } 
159   
160      private void updateDP() { 
161          mLdP = mL.getDesiredPower(); 
162          mRdP = mR.getDesiredPower(); 
163      } 
164   
165      private boolean driveDone() { 
166   
167          if (Math.abs(mL.getCurrentPosition()) > mL.getTargetPosition()) { 
168              mL.setDesiredPower(0); 
169          } 
170          if (Math.abs(mR.getCurrentPosition()) > mR.getTargetPosition()) { 
171              mR.setDesiredPower(0); 
172          } 
173          if (mL.getPower() < 0.01 && mR.getPower() < 0.01) { 
174              return true; 
175          } else { 
176              return false; 
177          } 
178      } 
179   
180      private boolean gyroDone() { 
181          if (Math.abs(gyro.getHeading() - turnTarget) < 3) { 
182              return true; 
183          } else { 
184              return false; 
185          } 
186      } 
187   
188      private boolean gyroEncoderDriveDone() { 
189          if (Math.abs(mL.getCurrentPosition()) < mL.getTargetPosition() && Math.abs(mR.getCurrentPosition()) < mR.getTargetPosition()) { 
190              gyroDrive(); 
191              return false; 
192          } else { 
193              return true; 
194          } 
195      } 
196   
197      private boolean gyroTimeDriveDone() { 
198          if (runTime.time() <= sleepTime) { 
199              gyroDrive(); 
200              return false; 
201          } else { 
202              return true; 
203          } 
204      } 
205   
206      private void gyroDrive() { 
207          if (gyro.getHeading() > 2 && gyro.getHeading() < 180) { 
208              setDrivePower(0.3, 0.5); 
209              skew = "RIGHT"; 
210          } else if (gyro.getHeading() < 358 && gyro.getHeading() > 180) { 
211              setDrivePower(0.5, 0.3); 
212              skew = "LEFT"; 
213          } else { 
214              setDrivePower(0.5); 
215              skew = "STRAIGHT"; 
216          } 
217      } 
218   
219      private boolean timeDone() { 
220          if (runTime.time() >= sleepTime) { 
221              return true; 
222          } else { 
223              return false; 
224          } 
225      } 
226   
227      private void driveStop() { 
228          mL.setDesiredPower(0); 
229          mR.setDesiredPower(0); 
230      }
