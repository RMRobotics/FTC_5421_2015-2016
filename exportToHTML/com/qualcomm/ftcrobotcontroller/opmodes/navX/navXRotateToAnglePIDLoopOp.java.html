31   package com.qualcomm.ftcrobotcontroller.opmodes.navX; 
32    
33   import com.kauailabs.navx.ftc.AHRS; 
34   import com.kauailabs.navx.ftc.navXPIDController; 
35   import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
36   import com.qualcomm.robotcore.hardware.DcMotor; 
37   import com.qualcomm.robotcore.hardware.DcMotorController; 
38   import com.qualcomm.robotcore.util.ElapsedTime; 
39    
40   /* 
41    * An example loop op mode where the robot will rotate 
42    * to a specified angle an then stop. 
43    * 
44    * This example uses a simple PID controller configuration 
45    * with a P coefficient, and will likely need tuning in order 
46    * to achieve optimal performance. 
47    * 
48    * Note that for the best accuracy, a reasonably high update rate 
49    * for the navX-Model sensor should be used. 
50    */ 
51   public class navXRotateToAnglePIDLoopOp extends OpMode { 
52       DcMotor leftMotor; 
53       DcMotor rightMotor; 
54    
55       /* This is the port on the Core Device Interface Module        */ 
56       /* in which the navX-Model Device is connected.  Modify this  */ 
57       /* depending upon which I2C port you are using.               */ 
58       private final int NAVX_DIM_I2C_PORT = 0; 
59       private AHRS navx_device; 
60       private navXPIDController yawPIDController; 
61       private ElapsedTime runtime = new ElapsedTime(); 
62    
63       private final byte NAVX_DEVICE_UPDATE_RATE_HZ = 50; 
64    
65       private final double TARGET_ANGLE_DEGREES = 90.0; 
66       private final double TOLERANCE_DEGREES = 2.0; 
67       private final double MIN_MOTOR_OUTPUT_VALUE = -1.0; 
68       private final double MAX_MOTOR_OUTPUT_VALUE = 1.0; 
69       private final double YAW_PID_P = 0.005; 
70       private final double YAW_PID_I = 0.0; 
71       private final double YAW_PID_D = 0.0; 
72    
73       private boolean first_iteration = true; 
74       navXPIDController.PIDResult yawPIDResult; 
75    
76       @Override 
77       public void init() { 
78           leftMotor = hardwareMap.dcMotor.get("left_drive"); 
79           rightMotor = hardwareMap.dcMotor.get("right_drive"); 
80    
81           navx_device = AHRS.getInstance(hardwareMap.deviceInterfaceModule.get("dim"), 
82                   NAVX_DIM_I2C_PORT, 
83                   AHRS.DeviceDataType.kProcessedData, 
84                   NAVX_DEVICE_UPDATE_RATE_HZ); 
85    
86           rightMotor.setDirection(DcMotor.Direction.REVERSE); 
87    
88           /* If possible, use encoders when driving, as it results in more */ 
89           /* predictable drive system response.                           */ 
90           leftMotor.setChannelMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
91           rightMotor.setChannelMode(DcMotorController.RunMode.RUN_USING_ENCODERS); 
92    
93           /* Create a PID Controller which uses the Yaw Angle as input. */ 
94           yawPIDController = new navXPIDController( navx_device, 
95                   navXPIDController.navXTimestampedDataSource.YAW); 
96    
97           /* Configure the PID controller */ 
98           yawPIDController.setSetpoint(TARGET_ANGLE_DEGREES); 
99           yawPIDController.setContinuous(true); 
100          yawPIDController.setOutputRange(MIN_MOTOR_OUTPUT_VALUE, MAX_MOTOR_OUTPUT_VALUE); 
101          yawPIDController.setTolerance(navXPIDController.ToleranceType.ABSOLUTE, TOLERANCE_DEGREES); 
102          yawPIDController.setPID(YAW_PID_P, YAW_PID_I, YAW_PID_D); 
103          yawPIDController.enable(true); 
104   
105          first_iteration = true; 
106      } 
107   
108      @Override 
109      public void loop() { 
110          /* reset the navX-Model device yaw angle so that whatever direction */ 
111          /* it is currently pointing will be zero degrees.                   */ 
112          if ( first_iteration ) { 
113              first_iteration = false; 
114              navx_device.zeroYaw(); 
115              yawPIDResult = new navXPIDController.PIDResult(); 
116          } 
117          /* Wait for new Yaw PID output values, then update the motors 
118             with the new PID value with each new output value. 
119           */ 
120   
121          final double TOTAL_RUN_TIME_SECONDS = 30.0; 
122          int DEVICE_TIMEOUT_MS = 500; 
123          navXPIDController.PIDResult yawPIDResult = new navXPIDController.PIDResult(); 
124   
125          if ( yawPIDController.isNewUpdateAvailable(yawPIDResult) ) { 
126              if ( yawPIDResult.isOnTarget() ) { 
127                  leftMotor.setPowerFloat(); 
128                  rightMotor.setPowerFloat(); 
129              } else { 
130                  double output = yawPIDResult.getOutput(); 
131                  if ( output < 0 ) { 
132                      /* Rotate Left */ 
133                      leftMotor.setPower(-output); 
134                      rightMotor.setPower(output); 
135                  } else { 
136                      /* Rotate Right */ 
137                      leftMotor.setPower(output); 
138                      rightMotor.setPower(-output); 
139                  } 
140              } 
141          } else { 
142              /* No sensor update has been received since the last time  */ 
143              /* the loop() function was invoked.  Therefore, there's no */ 
144              /* need to update the motors at this time.                 */ 
145          } 
146      } 
147  }
