32   package com.qualcomm.ftcrobotcontroller.opmodes.navX; 
33    
34   import com.kauailabs.navx.ftc.AHRS; 
35   import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
36   import com.qualcomm.robotcore.util.ElapsedTime; 
37    
38   import java.text.DecimalFormat; 
39    
40   /** 
41    *  navX-Micro Zero Yaw Op 
42    * <p> 
43    * Acquires processed data from navX-Micro 
44    * and displays it in the Robot DriveStation 
45    * as telemetry data.  This opmode demonstrates how to "zero" 
46    * (reset to zero degrees) the yaw.  This causes whatever 
47    * direction the navX-Model device is currently pointing to 
48    * now be zero degrees, and is an effective method for dealing 
49    * with accumulating Yaw Drift. 
50    * 
51    * For more information on Yaw drift, please see the online help at: 
52    * http://navx-micro.kauailabs.com/guidance/yaw-drift/ 
53    */ 
54   public class navXZeroYawOp extends OpMode { 
55    
56     /* This is the port on the Core Device Interace Module */ 
57     /* in which the navX-Micro is connected.  Modify this  */ 
58     /* depending upon which I2C port you are using.        */ 
59     private final int NAVX_DIM_I2C_PORT = 0; 
60    
61     private String startDate; 
62     private ElapsedTime runtime = new ElapsedTime(); 
63     private AHRS navx_device; 
64    
65     @Override 
66     public void init() { 
67       navx_device = AHRS.getInstance(hardwareMap.deviceInterfaceModule.get("dim"), 
68               NAVX_DIM_I2C_PORT, 
69               AHRS.DeviceDataType.kProcessedData); 
70     } 
71    
72     @Override 
73     public void stop() { 
74       navx_device.close(); 
75     } 
76     /* 
77        * Code to run when the op mode is first enabled goes here 
78        * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#start() 
79        */ 
80     @Override 
81     public void init_loop() { 
82       telemetry.addData("navX Op Init Loop", runtime.toString()); 
83     } 
84    
85     /* 
86      * This method will be called repeatedly in a loop 
87      * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#loop() 
88      */ 
89     @Override 
90     public void loop() { 
91    
92         boolean connected = navx_device.isConnected(); 
93         telemetry.addData("1 navX-Device", connected ? 
94                 "Connected" : "Disconnected" ); 
95         String gyrocal, magcal, yaw, pitch, roll, compass_heading; 
96         String fused_heading, ypr, cf, motion; 
97         DecimalFormat df = new DecimalFormat("#.##"); 
98    
99         if ( connected ) { 
100            gyrocal = (navx_device.isCalibrating() ? 
101                    "CALIBRATING" : "Calibration Complete"); 
102            magcal = (navx_device.isMagnetometerCalibrated() ? 
103                    "Calibrated" : "UNCALIBRATED"); 
104            yaw = df.format(navx_device.getYaw()); 
105            pitch = df.format(navx_device.getPitch()); 
106            roll = df.format(navx_device.getRoll()); 
107            ypr = yaw + ", " + pitch + ", " + roll; 
108            compass_heading = df.format(navx_device.getCompassHeading()); 
109            fused_heading = df.format(navx_device.getFusedHeading()); 
110            if (!navx_device.isMagnetometerCalibrated()) { 
111                compass_heading = "-------"; 
112            } 
113            cf = compass_heading + ", " + fused_heading; 
114            if ( navx_device.isMagneticDisturbance()) { 
115                cf += " (Mag. Disturbance)"; 
116            } 
117            motion = (navx_device.isMoving() ? "Moving" : "Not Moving"); 
118            if ( navx_device.isRotating() ) { 
119                motion += ", Rotating"; 
120            } 
121        } else { 
122            gyrocal = 
123              magcal = 
124              ypr = 
125              cf = 
126              motion = "-------"; 
127        } 
128        telemetry.addData("2 GyroAccel", gyrocal ); 
129        telemetry.addData("3 Y,P,R", ypr); 
130        telemetry.addData("4 Magnetometer", magcal ); 
131        telemetry.addData("5 Compass,9Axis", cf ); 
132        telemetry.addData("6 Motion", motion); 
133   
134        /* If the left 'bumper' button pressed, 
135           reset (zero) the current yaw angle.  This causes whatever 
136           direction the navX-Model device is currently pointing to 
137           now be zero degrees. 
138         */ 
139        //if ( gamepad1.left_bumper ) { 
140        if ( ( navx_device.getUpdateCount() % 500 ) == 0 ) { 
141          navx_device.zeroYaw(); 
142        } 
143    } 
144   
145  }
