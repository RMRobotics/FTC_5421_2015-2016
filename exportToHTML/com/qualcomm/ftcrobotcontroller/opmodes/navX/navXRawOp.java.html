40   /** 
41    * navXRawOp 
42    * <p> 
43    * This sample demonstrates how to acquire the raw 
44    * Gyroscope, Accelerometer and Magnetometer data.  This raw 
45    * data is typically not as useful as the "processed" data 
46    * (see the navXProcessedOp for details), however is provided 
47    * for those interested in accessing the raw data. 
48    * 
49    * Gyroscope data is units of Degrees/second. 
50    * Accelerometer data is in units of G. 
51    * Magnetometer data is in units if microTorr (uT) 
52    * 
53    * Magnetometer data is not valid unless magnetometer calibration 
54    * has been performed.  Without calibration, the magnetometer 
55    * data will be all zeros.  For details on how to calibrate the 
56    * magnetometer, please see the Magnetometer Calibration documentation: 
57    * http://navx-micro.kauailabs.com/guidance/magnetometer-calibration/ 
58    * 
59    * Note that due to limitations imposed by the Core Device 
60    * Interface Module's I2C interface mechanisms, the acquisition 
61    * of both processed data and raw data requires two separate 
62    * I2C bus transfers, and thus takes longer than acquiring 
63    * only the raw or only the processed data. 
64    */ 
65   public class navXRawOp extends OpMode { 
66    
67     private final int NAVX_DIM_I2C_PORT = 0; 
68     private String startDate; 
69     private ElapsedTime runtime = new ElapsedTime(); 
70     private AHRS navx_device; 
71     @Override 
72     public void init() { 
73         navx_device = AHRS.getInstance(hardwareMap.deviceInterfaceModule.get("dim"), 
74                 NAVX_DIM_I2C_PORT, 
75                 AHRS.DeviceDataType.kQuatAndRawData);  } 
76    
77     @Override 
78     public void stop() { 
79       navx_device.close(); 
80     } 
81     /* 
82        * Code to run when the op mode is first enabled goes here 
83        * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#start() 
84        */ 
85     @Override 
86     public void init_loop() { 
87       telemetry.addData("navX Op Init Loop", runtime.toString()); 
88     } 
89    
90     /* 
91      * This method will be called repeatedly in a loop 
92      * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#loop() 
93      */ 
94     @Override 
95     public void loop() { 
96    
97         boolean connected = navx_device.isConnected(); 
98         telemetry.addData("1 navX-Device", connected ? "Connected" : "Disconnected" ); 
99         String gyrocal, gyro_raw, accel_raw, mag_raw; 
100        boolean magnetometer_calibrated; 
101        if ( connected ) { 
102            DecimalFormat df = new DecimalFormat("#.##"); 
103            magnetometer_calibrated = navx_device.isMagnetometerCalibrated(); 
104            gyro_raw = df.format(navx_device.getRawGyroX()) + ", " + 
105                        df.format(navx_device.getRawGyroY()) + ", " + 
106                        df.format(navx_device.getRawGyroZ()); 
107            accel_raw = df.format(navx_device.getRawAccelX()) + ", " + 
108                        df.format(navx_device.getRawAccelY()) + ", " + 
109                        df.format(navx_device.getRawAccelZ()); 
110            if ( magnetometer_calibrated ) { 
111                mag_raw = df.format(navx_device.getRawMagX()) + ", " + 
112                        df.format(navx_device.getRawMagY()) + ", " + 
113                        df.format(navx_device.getRawMagZ()); 
114            } else { 
115                mag_raw = "Uncalibrated"; 
116            } 
117        } else { 
118          gyro_raw = 
119              accel_raw = 
120              mag_raw = "-------"; 
121        } 
122        telemetry.addData("2 Gyros (Degrees/Sec):", gyro_raw); 
123        telemetry.addData("3 Accelerometers  (G):", accel_raw ); 
124        telemetry.addData("4 Magnetometers  (uT):", mag_raw ); 
125    }
