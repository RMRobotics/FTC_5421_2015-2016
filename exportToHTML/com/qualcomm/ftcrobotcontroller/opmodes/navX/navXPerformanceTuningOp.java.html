32   package com.qualcomm.ftcrobotcontroller.opmodes.navX; 
33    
34   import com.kauailabs.navx.ftc.AHRS; 
35   import com.kauailabs.navx.ftc.navXPerformanceMonitor; 
36   import com.qualcomm.robotcore.eventloop.opmode.OpMode; 
37   import com.qualcomm.robotcore.util.ElapsedTime; 
38    
39   import java.text.DecimalFormat; 
40    
41   /** 
42    * navX-Micro Performance Tuning Op 
43    * 
44    * This opmode provides insight into the peformance of the communication 
45    * with the navX-Model sensor over the I2C bus via the Core Device Interface 
46    * Module.  Since the Android operating system is not a real-time 
47    * operating system, and since communication w/the navX-Model sensor is 
48    * occurring over a wifi-direct network which can be prone to interference, 
49    * the actual performance (update rate) achieved may be less than 
50    * one might expect. 
51    * 
52    * Since the navX-Model devices integrate sensor data onboard, to achieve 
53    * the best performance for device control methods like a PID controller 
54    * that work best with constant-time updates, the strategy is to: 
55    * 
56    * 1) Configure the navX-Model device to a high update rate (e.g., 50Hz) 
57    * 2) Using this performance-tuning Op-Mode (with all other 
58    * sensors connected, just as your robot will be configured for normal 
59    * use) observe the "effective" update rate, and the number of missed 
60    * samples over the last minute. 
61    * 3) Lower the navX-Model device update rate until the number of missed 
62    * samples over the last minute reaches zero. 
63    */ 
64   public class navXPerformanceTuningOp extends OpMode { 
65    
66     /* This is the port on the Core Device Interface Module */ 
67     /* in which the navX-Micro is connected.  Modify this  */ 
68     /* depending upon which I2C port you are using.        */ 
69     private final int NAVX_DIM_I2C_PORT = 0; 
70    
71     private AHRS navx_device; 
72     private navXPerformanceMonitor navx_perfmon; 
73     private byte sensor_update_rate_hz = 40; 
74     private ElapsedTime runtime = new ElapsedTime(); 
75    
76     @Override 
77     public void init() { 
78       AHRS.setLogging(true); 
79       navx_device = AHRS.getInstance(hardwareMap.deviceInterfaceModule.get("dim"), 
80               NAVX_DIM_I2C_PORT, 
81               AHRS.DeviceDataType.kProcessedData, 
82               sensor_update_rate_hz); 
83       navx_perfmon = new navXPerformanceMonitor(navx_device); 
84     } 
85    
86   @Override 
87     public void start() { 
88       navx_device.registerCallback(navx_perfmon); 
89   } 
90    
91     @Override 
92     public void stop() { 
93       navx_device.close(); 
94     } 
95     /* 
96        * Code to run when the op mode is first enabled goes here 
97        * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#start() 
98        */ 
99     @Override 
100    public void init_loop() { 
101      telemetry.addData("navX Op Init Loop", runtime.toString()); 
102    } 
103   
104    /* 
105     * This method will be called repeatedly in a loop 
106     * @see com.qualcomm.robotcore.eventloop.opmode.OpMode#loop() 
107     */ 
108    @Override 
109    public void loop() { 
110   
111        boolean connected = navx_device.isConnected(); 
112        telemetry.addData("1 navX-Device...............:", connected ? 
113                "Connected" : "Disconnected" ); 
114        String gyrocal, motion; 
115        DecimalFormat df = new DecimalFormat("#.##"); 
116   
117        telemetry.addData("2 Sensor Rate (Hz)...", Byte.toString(navx_device.getActualUpdateRate())); 
118        telemetry.addData("3 Transfer Rate (Hz).", Integer.toString(navx_device.getCurrentTransferRate())); 
119        telemetry.addData("4 Delivvered Rate (Hz)", Integer.toString(navx_perfmon.getDeliveredRateHz())); 
120        telemetry.addData("5 Missed Samples.....", Integer.toString(navx_perfmon.getNumMissedSensorTimestampedSamples())); 
121        telemetry.addData("6 Duplicate Samples..", Integer.toString(navx_device.getDuplicateDataCount())); 
122        telemetry.addData("7 Sensor deltaT (ms).", Long.toString(navx_perfmon.getLastSensorTimestampDeltaMS())); 
123        telemetry.addData("8 System deltaT (ms).", Long.toString(navx_perfmon.getLastSystemTimestampDeltaMS())); 
124    } 
125  }
